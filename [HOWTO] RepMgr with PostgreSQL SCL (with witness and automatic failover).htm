[HOWTO] RepMgr with PostgreSQL SCL (with witness and automatic failover)
 

[HOWTO] RepMgr with PostgreSQL SCL (with witness and automatic failover)

​########################################
# HOWTO use REPMGR with REDHAT 9.5 SCL #
# With automatic failover (witness + repmgrd)
########################################

# ONLINE README : https://github.com/2ndQuadrant/repmgr

# REPMGR VERSION   : 3.3
# REDHAT VERSION   : 7.2
# POSTGRES VERSION : 9.5 SCL

# DATABASE : dmbadpoc
# NODE 1   : dmbadpoc_hot0 (master)
# NODE 2   : dmbadpoc_hot1 (slave)
# NODE 3   : dmbadpoc_witness

# required package before compiling :
rh-postgresql95-postgresql-devel.x86_64
rh-postgresql95-postgresql-static.x86_64
libxslt-devel.x86_64
libxml2-devel.x86_64
pam-devel.x86_64
openssl-devel
libselinux-devel
readline-devel
# and in postgres.conf :
wal_log_hints = on
shared_preload_libraries = 'pg_stat_statements, repmgr_funcs'

# Download repmgr source code from : http://www.repmgr.org/download/repmgr-3.2.1.tar.gz
# To make repmgr able to make remote commands with the postgreSQL SCL version you need to use modified source by Eric DELEFORTERIE (repmgr-master-DELE-3.3dev.zip)
# Unzip in /produits/pgsql/repmgr-master

 

make USE_PGXS=1
sudo make USE_PGXS install OR copy manually (ask IS TEAM to do it for you as "root")

cp repmgr /opt/rh/rh-postgresql94/root/usr/bin/repmgr
cp remgrd /opt/rh/rh-postgresql94/root/usr/bin/repmgrd
cp sql/repmgr_funcs.so /opt/rh/rh-postgresql94/root/usr/lib64/pgsql/repmgr_funcs.so

# switchover from the standby
# needs wal_log_hints to be enabled before dbinit
. /produits/pgsql/env/env_dmbadpoc_hot1
repmgr -f $BASE/admin/repmgr.conf standby switchover

# failover
. /produits/pgsql/env/env_dmbadpoc_hot0
pg_ctl -m fast stop
. /produits/pgsql/env/env_dmbadpoc_hot1
repmgr -f $BASE/admin/repmgr.conf standby promote

# NODE 1 (master)
# ENV FILE :
. /produits/pgsql/env/env_dmbadpoc_hot0
###
. /produits/pgsql/env_95
export DBNAME=dmbadpoc
export DBHOST=dmbalx02
export PGPORT=12400
export BASE=/bases/pgsql/$DBNAME
export PGDATA=$BASE/data/pg
export PGLOG=$BASE/var/log
export PS1="$(whoami)@$(hostname):$DBNAME:\$PWD # HOT_STANDBY # "
###

createuser -s repmgr
createdb repmgr -O repmgr

vi $PGDATA/pg_hba.conf
# Add these lines in the top of pg_hba.conf file
local   replication   repmgr                              trust
host    replication   repmgr      127.0.0.1/32            trust
host    replication   repmgr      192.88.67.74/32          trust

local   repmgr        repmgr                              trust
host    repmgr        repmgr      127.0.0.1/32            trust
host    repmgr        repmgr      192.88.67.74/32          trust


vi $PGDATA/postgresql.conf # no need to use rsync or scp in the archive_command parameter with repmgr
# Add this line (replace wal_keep_segments if exists)
max_replication_slots = 5


pg_ctl restart


vi $BASE/admin/repmgr.conf
####
cluster=dmbadpoc
node=1
node_name=dmbadpoc_hot0
conninfo='host=dmbalx02 port=12400 user=repmgr dbname=repmgr'
pg_bindir=/opt/rh/rh-postgresql95/root/usr/bin
remote_command_prefix=scl enable rh-postgresql95 --
use_replication_slots = 1
failover=automatic
logfile=/bases/pgsql/dmbadpoc/var/log/repmgr.log
promote_command='repmgr standby promote -f /bases/pgsql/dmbadpoc/admin/repmgr.conf --log-to-file'
follow_command='repmgr standby follow -f /bases/pgsql/dmbadpoc/admin/repmgr.conf --log-to-file'
####

psql -d repmgr
    ALTER USER repmgr SET search_path TO repmgr_dmbadpoc, "$user", public;

repmgr -f $BASE/admin/repmgr.conf master register
    NOTICE: master node correctly registered for cluster dmbadpoc with id 1 (conninfo: host=dmbalx02 port=12400 user=repmgr dbname=repmgr)
   
psql -U repmgr -d repmgr
    SELECT * FROM repl_nodes;

   

# NODE 2 (slave)
# ENV FILE :
. /produits/pgsql/env/env_dmbadpoc_hot1
##
. /produits/pgsql/env_95
export DBNAME=dmbaipoc
export DBHOST=dmbalx02
export PGPORT=12401
export BASE=/bases/pgsql/$DBNAME
export PGDATA=$BASE/data/pg
export PGLOG=$BASE/var/log
export PS1="$(whoami)@$(hostname):$DBNAME:\$PWD # HOT_STANDBY # "
##

vi $BASE/admin/repmgr.conf
##
cluster=dmbadpoc
node=2
node_name=dmbadpoc_hot1
conninfo='host=dmbalx02 port=12401 user=repmgr dbname=repmgr'
pg_bindir=/opt/rh/rh-postgresql95/root/usr/bin
remote_command_prefix=scl enable rh-postgresql95 --
use_replication_slots = 1
failover=automatic
logfile=/bases/pgsql/dmbaipoc/var/log/repmgr.log
promote_command='repmgr standby promote -f /bases/pgsql/dmbaipoc/admin/repmgr.conf --log-to-file'
follow_command='repmgr standby follow -f /bases/pgsql/dmbaipoc/admin/repmgr.conf --log-to-file'
##

# CLONE master to slave
rm -rf $PGDATA/*
repmgr -h dmbalx02 -U repmgr -d repmgr -p 12400 -D $PGDATA -f $BASE/admin/repmgr.conf standby clone

# copy *.conf from master, modify config files (ip, files)
vi $PGDATA/postgresql.conf
vi $PGDATA/pg_hba.conf

# start slave
pg_ctl start

# on the master connect as repmgr
psql -U repmgr -d repmgr
repmgr=# \x on
repmgr=# SELECT * FROM pg_stat_replication;
-[ RECORD 1 ]----+------------------------------
pid              | 4420
usesysid         | 24577
usename          | repmgr
application_name | dmbadpoc_hot1
client_addr      | 192.88.67.74
client_hostname  |
client_port      | 41958
backend_start    | 2016-12-20 17:07:24.858867+01
backend_xmin     |
state            | streaming
sent_location    | 0/B0001A8
write_location   | 0/B0001A8
flush_location   | 0/B0001A8
replay_location  | 0/B0001A8
sync_priority    | 0
sync_state       | async


# register the slave
. /produits/pgsql/env/env_dmbadpoc_hot1
repmgr -f $BASE/admin/repmgr.conf standby register
    NOTICE: standby node correctly registered for cluster dmbadpoc with id 2 (conninfo: host=dmbalx02 port=12401 user=repmgr dbname=repmgr)

# check repl_nodes   
psql -U repmgr -d repmgr   
repmgr=# SELECT * FROM repl_nodes ORDER BY id;   
# or   repmgr -f $BASE/admin/repmgr.conf cluster show 
ps -ef|grep postgres # compare xlog number


# test
. /produits/pgsql/env/env_dmbadpoc_hot0
psql
\c dmbapoc
CREATE table bb AS SELECT generate_series(1,1000000);

. /produits/pgsql/env/env_dmbadpoc_hot1
psql
\c dmbapoc
select * from bb;


####
# CREATE WITNESS
####

# ENV FILE :
. /produits/pgsql/env/env_dmbadpoc_witness
##
. /produits/pgsql/env_95
export DBNAME=dmbawpoc
export DBHOST=dmbalx02
export PGPORT=12402
export BASE=/bases/pgsql/$DBNAME
export PGDATA=$BASE/data/pg
export PGLOG=$BASE/var/log
export PS1="$(whoami)@$(hostname):$DBNAME:\$PWD # WITNESS # "
##

vi $BASE/admin/repmgr.conf
##
cluster=dmbadpoc
node=3
node_name=dmbadpoc_witness
conninfo='host=dmbalx02 port=12402 user=repmgr dbname=repmgr'
pg_bindir=/opt/rh/rh-postgresql95/root/usr/bin
remote_command_prefix=scl enable rh-postgresql95 --
use_replication_slots = 1
failover=automatic
logfile=/bases/pgsql/dmbadpoc/dmbawpoc/var/log/repmgr.log
promote_command='repmgr standby promote -f /bases/pgsql/dmbadpoc/dmbawpoc/admin/repmgr.conf --log-to-file'
follow_command='repmgr standby follow -f /bases/pgsql/dmbadpoc/dmbawpoc/admin/repmgr.conf --log-to-file'
##

# create witness
. /produits/pgsql/env/env_dmbadpoc_witness
rm -rf $PGDATA/*
repmgr -h dmbalx02 -U repmgr -d repmgr -p 12400 -D $PGDATA -f $BASE/admin/repmgr.conf witness create

# For repmgr 4.1

initdb --data-checksums --encoding=UTF8 --locale=fr_FR.utf8

pg_ctl start

# End repmgr 4.1​ ​
# copy *.conf from master, modify config files (ip, files)
vi $PGDATA/postgresql.conf
vi $PGDATA/pg_hba.conf

# start slave
pg_ctl restart

# For repmgr 4.1

createuser -s repmgr

createdb repmgr -O repmgr​

​​psql -d repmgr

ALTER USER repmgr SET search_path TO repmgr_<PGNAME>, "$user", public;

# End repmgr 4.1
# register witness
repmgr -h dmbalx02 -U repmgr -d repmgr -p 12400 -D $PGDATA -f $BASE/admin/repmgr.conf witness register
NOTICE: configuration has been successfully copied to the witness

# show cluster
repmgr -f $BASE/admin/repmgr.conf cluster show 


Role      | Name             | Upstream      | Connection String
----------+------------------|---------------|---------------------------------------------------
* master  | dmbadpoc_hot0    |               | host=dmbalx02 port=12400 user=repmgr dbname=repmgr
  standby | dmbadpoc_hot1    | dmbadpoc_hot0 | host=dmbalx02 port=12401 user=repmgr dbname=repmgr
  witness | dmbadpoc_witness |               | host=dmbalx02 port=12402 user=repmgr dbname=repmgr

# start repmgrd daemon on every node
. /produits/pgsql/env/env_dmbadpoc_hot0
repmgrd -d -p $BASE/var/repmgr.pid -f $BASE/admin/repmgr.conf &

. /produits/pgsql/env/env_dmbadpoc_hot1
repmgrd -d -p $BASE/var/repmgr.pid -f $BASE/admin/repmgr.conf &

. /produits/pgsql/env/env_dmbadpoc_witness
repmgrd -d -p $BASE/var/repmgr.pid -f $BASE/admin/repmgr.conf &

# test automatic failover
. /produits/pgsql/env/env_dmbadpoc_hot0
pg_ctl stop

. /produits/pgsql/env/env_dmbadpoc_hot1
repmgr -f $BASE/admin/repmgr.conf cluster show 

Role      | Name             | Upstream | Connection String
----------+------------------|----------|---------------------------------------------------
  witness | dmbadpoc_witness |          | host=dmbalx02 port=12402 user=repmgr dbname=repmgr
  FAILED  | dmbadpoc_hot0    |          | host=dmbalx02 port=12400 user=repmgr dbname=repmgr
* master  | dmbadpoc_hot1    |          | host=dmbalx02 port=12401 user=repmgr dbname=repmgr


# after automatic failover, to switchback to the old master
. /produits/pgsql/env/env_dmbadpoc_hot0
pg_ctl stop
rm -rf $PGDATA/*

repmgr --force -h dmbalx02 -U repmgr -d repmgr -p 12401 -D $PGDATA -f $BASE/admin/repmgr.conf standby clone
# change ip in $PGDATA/postgresql.conf then start
pg_ctl start
# register standby
repmgr --force -f $BASE/admin/repmgr.conf standby register
repmgr -f $BASE/admin/repmgr.conf cluster show 

Role      | Name             | Upstream      | Connection String
----------+------------------|---------------|---------------------------------------------------
  witness | dmbadpoc_witness |               | host=dmbalx02 port=12402 user=repmgr dbname=repmgr
* master  | dmbadpoc_hot1    |               | host=dmbalx02 port=12401 user=repmgr dbname=repmgr
  standby | dmbadpoc_hot0    | dmbadpoc_hot1 | host=dmbalx02 port=12400 user=repmgr dbname=repmgr

 
repmgrd -d -p $BASE/var/repmgr.pid -f $BASE/admin/repmgr.conf &


repmgr -f $BASE/admin/repmgr.conf cluster show 
repmgr -f $BASE/admin/repmgr.conf standby switchover

Role      | Name             | Upstream      | Connection String
----------+------------------|---------------|---------------------------------------------------
  witness | dmbadpoc_witness |               | host=dmbalx02 port=12402 user=repmgr dbname=repmgr
  standby | dmbadpoc_hot1    |               | host=dmbalx02 port=12401 user=repmgr dbname=repmgr
* master  | dmbadpoc_hot0    | dmbadpoc_hot0 | host=dmbalx02 port=12400 user=repmgr dbname=repmgr

# force re-register master & standby
. /produits/pgsql/env/env_dmbadpoc_hot0
repmgr -F -f $BASE/admin/repmgr.conf master register


. /produits/pgsql/env/env_dmbadpoc_hot1
repmgr -F -f $BASE/admin/repmgr.conf standby register

# restart repmgrd daemon on every node
. /produits/pgsql/env/env_dmbadpoc_hot0
repmgrd -d -p $BASE/var/repmgr.pid -f $BASE/admin/repmgr.conf &

. /produits/pgsql/env/env_dmbadpoc_hot1
repmgrd -d -p $BASE/var/repmgr.pid -f $BASE/admin/repmgr.conf &

repmgrd -d -p $BASE/var/repmgr.pid -f $BASE/admin/repmgr.conf &
